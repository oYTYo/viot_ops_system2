# backend/models.py 新增核心代码
from sqlalchemy import Table, Column, String, ForeignKey

# 分组与摄像机的多对多关联表
group_camera_link = Table(
    "group_camera_link",
    Base.metadata,
    Column("group_id", String(64), ForeignKey("device_group.id", ondelete="CASCADE"), primary_key=True),
    Column("camera_id", String(64), ForeignKey("camera.id", ondelete="CASCADE"), primary_key=True),
)

class DeviceGroup(Base):
    """设备自定义分组"""
    __tablename__ = "device_group"

    id: Mapped[str] = mapped_column(String(64), primary_key=True)
    name: Mapped[str] = mapped_column(String(64), nullable=False)
    parent_id: Mapped[str | None] = mapped_column(String(64), ForeignKey("device_group.id", ondelete="CASCADE"), nullable=True)
    sort_order: Mapped[int] = mapped_column(Integer, default=0)

    # 关联摄像机
    cameras: Mapped[list["Camera"]] = relationship(
        secondary=group_camera_link,
        backref="groups",
        lazy="selectin"
    )
2. 新增分组 API (backend/main.py)
需要提供新建分组（同时绑定摄像机）、删除分组、以及获取分组树的接口。

Python
# backend/main.py 核心逻辑
@app.post("/groups")
def create_group(payload: schemas.DeviceGroupCreate, db: Session = Depends(get_db)):
    # 1. 创建分组基本信息
    new_group = models.DeviceGroup(
        id=generate_id(), 
        name=payload.name, 
        parent_id=payload.parent_id
    )
    # 2. 绑定选中的摄像机 (假设 payload 传了 camera_ids 列表)
    if payload.camera_ids:
        cameras = db.query(models.Camera).filter(models.Camera.id.in_(payload.camera_ids)).all()
        new_group.cameras.extend(cameras)
        
    db.add(new_group)
    db.commit()
    return new_group

@app.delete("/groups/{group_id}")
def delete_group(group_id: str, db: Session = Depends(get_db)):
    # 级联删除处理（根据 parent_id 关联关系和级联设置）
    group = db.get(models.DeviceGroup, group_id)
    db.delete(group)
    db.commit()
    return {"message": "deleted"}
